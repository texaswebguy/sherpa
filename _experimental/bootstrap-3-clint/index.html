  1.      $('[data-toggle="popover"]').popover({trigger:'manual'});
  2. $('[data-toggle="popover"]').click(function(event){
  3.   event.preventDefault();
  4.   $('[data-toggle="popover"]').popover('destroy');
  5.   $(this).popover('show');
  6.   $('[data-dismiss="popover"]').click(function(event){
  7.     event.preventDefault();
  8.     $(this).parents('.popover').prev().popover('hide');
  9.   });
  10. });
